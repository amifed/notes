# 背包问题

背包问题是很常见的动态规划问题

# 01 背包

NNN 个物品，容量为 MMM 的背包，第 iii 件物品的体积为 viv_ivi​，价值为 wiw_iwi​ 。每件物品最多只能用一次，求背包能装下的物品的价值和的最大值？

# 思路分析

定义 f(i,j)f(i, j)f(i,j) 为从前 iii 个物品中选择体积不超过 jjj 的物品的价值和的最大值，则 f(N,M)f(N, M)f(N,M) 即为从 NNN 个物品中选择体积不超过 VVV 的物品的价值和的最大值。

对于 f(i,j)f(i, j)f(i,j)，即第 iii 个物品，我们可以选择放入背包 000 或 111 次（即放入背包和不放入背包）：

  • 放入背包 000 次（包括背包容量放不下当前物品）：那 f(i,j)f(i, j)f(i,j) 应该为前 i−1i - 1i−1 个物品中体积不超过 jjj 的物品的价值和的最大值，即 f(i,j)=f(i−1,j)f(i, j) = f(i - 1, j)f(i,j)=f(i−1,j)；
  • 放入背包 111 次：设第 iii 个物品的体积为 viv_ivi​。如果要将第 iii 个物品放入背包，那此时背包剩余容量应该不超过 j−vij - v_ij−vi​，那 f(i,j)f(i, j)f(i,j) 应该为前 i−1i - 1i−1 个物品中体积不超过 j−vij - v_ij−vi​ 的物品的价值和的最大值加上第 iii 个物品的体积，即 f(i,j)=f(i−1,j−vi)+w[i]f(i,j) = f(i-1,j-v_i)+w[i]f(i,j)=f(i−1,j−vi​)+w[i]；

最后讨论完所有情况，f(i,j)f(i, j)f(i,j) 为前 iii 个物品中选择体积不超过 jjj 的物品的价值和的最大值，所以 f(i,j)f(i, j)f(i,j) 应对两种情况求最大值。

边界条件

当 i=0i=0i=0 时，000 个物品，不管背包容量为多少，最大值都应为 000，即 f(0,j)=0,0≤j≤m;f(0,j)=0,0≤j≤m;f(0,j)=0,0≤j≤m; 当 j=0j=0j=0 时，背包容量为 000，无法放下任何物品，最大值都应为 000，即 f(i,0)=0,0≤i≤n.f(i,0)=0,0≤i≤n.f(i,0)=0,0≤i≤n.

公式

f(i,j)=f(i−1,j),1≤i≤n,0≤j<vi;f(i,j)=f(i-1,j),1≤i≤n,0≤j<v_i;f(i,j)=f(i−1,j),1≤i≤n,0≤j<vi​;

f(i,j)=max{f(i−1,j),f(i−1,j−vi)+w[i]},1≤i≤n,vi≤j≤m;f(i,j)=max\{f(i-1,j),f(i-1,j-v_i)+w[i]\},1≤i≤n,v_i≤j≤m;f(i,j)=max{f(i−1,j),f(i−1,j−vi​)+w[i]},1≤i≤n,vi​≤j≤m;

# 参考代码

#include <iostream>

using namespace std;
const int N = 1e3 + 7;

int n, m;
int v[N], w[N];
int f[N][N];

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) cin >> v[i] >> w[i];
    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j <= m; ++j) {
            f[i][j] = f[i - 1][j];
            if (j >= v[i]) f[i][j] = max(f[i][j], f[i - 1][j - v[i]] + w[i]);
        }
    }
    cout << f[n][m];
    return 0;
}

# 空间优化

由上面分析可以，每次对第 iii 个物品计算时，都只依赖于第 i−1i - 1i−1 个物品的值，而二维数组 f[N][N] 保存的是第 0∼n0 \sim n0∼n 个物品的值， 也就是在计算时，数组第 iii 行的值都是通过第 i−1i - 1i−1 计算得出。因此在只需求出最终结果的前提下，不需要保存所有的值，可以将二维数组优化为一维数组（每次只保存上一行的数据），这种优化也可以理解为代码的等价变形。

当外层循环进行第 iii 轮时，一维数组内保存的为第 i−1i - 1i−1 轮的值，可以在不影响计算的情况下直接对其修改。

f[i][j] = f[i - 1][j]可以理解为第 iii 行第 jjj 列的值等于第 i−1i -1i−1 行第 jjj 列的值，可以等价修改为 f[j] = f[j]，又这是一个恒等式，可以将其删除。

循环变为

for (int i = 1; i <= n; ++i) {
  for (int j = 0; j <= m; ++j) {
    if (j >= v[i]) f[i][j] = max(f[i][j], f[i - 1][j - v[i]] + w[i]);
  }
}

而将内层语句修改为一维，第 iii 轮求值依赖于第 i−1i - 1i−1 轮的结果，要求改一维必须保证不会得到错误的第 i−1i -1i−1 轮的值，也就是说如果在 第 i,ji, ji,j 轮要用到第 i−1,vi−1∼mi-1,v_{i-1} \sim mi−1,vi−1​∼m 轮的值，此时一维数组对 f[j]表示第 i−1,ji-1,ji−1,j 轮的值，当该轮对 f[j] 做了修改，当第 i,j+1i, j + 1i,j+1 轮时，f[j] 已经不是第 i−1,ji - 1,ji−1,j 轮的值了，那计算将会出错。

这就在运用滚动数组对空间优化时要考虑的问题，在某一轮对数组的修改会不会覆盖掉上一轮的值，而导致本轮的计算取到的是错误的值（并非上一轮的值）。

总结一句就是，在正向遍历（下标从小到大）时，对其进行修改（覆盖旧值）后，在之后的遍历中仍会用到较小下标的旧值。

所以解决方案就是反向遍历（下标从大到小），先对大的值修改，之后的遍历只会用到更小的下标的旧值，而它们还没有被修改

for (int i = 1; i <= n; ++i) {
  for (int j = m; j >= v[i]; ++j) {
		f[j] = max(f[j], f[j - v[i]] + w[i]);
  }
}

#include <iostream>

using namespace std;
const int N = 1e3 + 7;

int n, m;
int f[N];

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) {
        int v, w;
        cin >> v >> w;
        for (int j = m; j >= v; --j) {
            f[j] = max(f[j], f[j - v] + w);
        }
    }
    cout << f[m];
    return 0;
}

# 多重背包

# 思路分析

跟完全背包相似，对于第 iii 件物品，最多只能放入背包 s[i]s[i]s[i] 次

状态转移方程

f(i,j)=max{f(i−1,j),f(i−1,j−vi)+w[i],f(i−1,j−2vi)+2w[i],⋯,f(i−1,j−sivi)+siw[i]},1≤i≤n,j−sivi≥0;f(i,j)=max\{f(i-1,j),f(i-1,j-v_i)+w[i],f(i-1,j-2v_i)+2w[i],\cdots,f(i-1,j-s_iv_i)+s_iw[i]\},1≤i≤n,j-s_iv_i≥0;f(i,j)=max{f(i−1,j),f(i−1,j−vi​)+w[i],f(i−1,j−2vi​)+2w[i],⋯,f(i−1,j−si​vi​)+si​w[i]},1≤i≤n,j−si​vi​≥0;

#include <iostream>
using namespace std;

const int  N = 107;
int n, m;

}

# 二进制优化

将 sss 使用 20,21,22,23,24,⋯2^0, 2^1, 2^2, 2^3, 2^4, \cdots20,21,22,23,24,⋯ 进行分解，可得到公式 s=20+21+22+23+⋯+2k+Cs=2^0+2^1+2^2+2^3+\cdots+2^k+Cs=20+21+22+23+⋯+2k+C；

对分解后的数组使用 01 背包的方式求解

#include <iostream>
using namespace std;

const int N = 2e5 + 7;

int n, m;
int v[N], w[N];
int f[N];

int main() {
    cin >> n >> m;
    int cnt = 0;
    for (int i = 1; i <= n; ++i) {
        int a, b, s;
        cin >> a >> b >> s;
        int k = 1;
        while (k <= s) {
            ++cnt;
            v[cnt] = a * k, w[cnt] = b * k;
            s -= k, k *= 2;
        }
        if (s > 0) {
            ++cnt;
            v[cnt] = a * s, w[cnt] = b * s;
        }
    }
    for (int i = 1; i <= cnt; ++i) {
        for (int j = m; j >= v[i]; --j) {
            f[j] = max(f[j], f[j - v[i]] + w[i]);
        }
    }
    cout << f[m];
    return 0;
}

#include <iostream>

# 01 背包问题求方案数

根据 01 背包求解的过程中，记录当前选择的方案数，将最优方案数累加（需注意两种选择相等的情况），最后根据最优方案，统计所有方案数；

#include <iostream>
#include <algorithm>
using namespace std;

const int N = 1e3 + 7;
const int MOD = 1e9 + 7;

int f[N], g[N];
int v, w;
int n, m;

int main() {
    cin >> n >> m;
    g[0] = 1;
    for (int i = 1; i <= n; ++i) {
        cin >> v >> w;
        for (int j = m; j >= v; --j) {
            int t = max(f[j], f[j - v] + w);
            int s = 0;
            if (t == f[j]) s = (s + g[j]) % MOD;
            if (t == f[j - v] + w) s = (s + g[j - v]) % MOD;
            f[j] = t, g[j] = s;
        }
    }
    int ma = *max_element(f + 1, f + m + 1);
    int ans = 0;
    for (int i = 1; i <= m; ++i) {
        if (f[i] == ma) ans = (ans + g[i]) % MOD;
    }
    cout << ans;
    return 0;
}

# 完全背包求方案数

#include <iostream>
using namespace std;

const int N = 1e4 + 7;
long long f[N];
long long v, n, m;

int main() {
    cin >> n >> m;
    f[0] = 1;
    for (int i = 0; i < n; ++i) {
        cin >> v;
        for (int j = v; j <= m; ++j) {
            f[j] += f[j - v];
        }
    }
    cout << f[m];
    return 0;
}
