对于 f(i,j)f(i, j)f(i,j)，即第 iii 个数，我们可以选择使用 0,1,2,⋯,k0, 1, 2, \cdots, k0,1,2,⋯,k 次参与划分：

  • …\dots…

# 朴素写法

#include <iostream>
using namespace std;

const int N = 1e3 + 7;
const int MOD = 1e9 + 7;

int n;
int f[N][N];

int main() {
    cin >> n;
    f[0][0] = 1;
    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j <= n; ++j) {
            for (int k = 0; k * i <= j; ++k) {
                f[i][j] = (f[i][j] + f[i - 1][j - k * i]) % MOD;
            }
        }
    }
    cout << f[n][n];
    return 0;
}

#include <iostream>
using namespace std;

const int N = 1e3 + 7;
const int MOD = 1e9 + 7;

int f[N][N];

int main() {
    cin >> n;
    f[0][0] = 1;
    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j <= n; ++j) {
            f[i][j] = (f[i][j] + f[i - 1][j]) % MOD;
            if (j >= i) f[i][j] = (f[i][j] + f[i][j - i]) % MOD;
        }
    }
    cout << f[n][n];
    return 0;
}

# 解法二

另一种定义状态的方式，

f[i][j]f[i][j]f[i][j] 表示总和是 iii ，且恰好表示成 jjj 个数的和的方案总数；

状态可分为两种情况，划分方案中包含 111 和 不包含 111。

状态转移方程： f[i][j]=f[i−1][j−1]+f[i−j][j]f[i][j]=f[i-1][j-1]+f[i-j][j]f[i][j]=f[i−1][j−1]+f[i−j][j]； 最终方案数为 ∑i=1nf[n][i]\displaystyle\sum_{i=1}^n f[n][i]i=1∑n​f[n][i]。

#include <iostream>
using namespace std;

const int N = 1e3 + 7;
const int MOD = 1e9 + 7;

int n;
int f[N][N];

int main() {
    cin >> n;
    f[0][0] = 1;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= i; ++j) {
            f[i][j] = (f[i - 1][j - 1] + f[i - j][j]) % MOD;
        }
    }
    int ans = 0;
    for (int i = 1; i <= n; ++i) ans = (ans + f[n][i]) % MOD;
    cout << ans;
    return 0;
}
