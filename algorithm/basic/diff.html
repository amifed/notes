# 差分数列

# 定义

数列 aaa, bbb 满足以下公式：

b1=a1{b_1 = a_1}b1​=a1​

b2=a2−a1b_2 = a_2 - a_1b2​=a2​−a1​

b3=a3−a2b_3 = a_3 - a_2b3​=a3​−a2​

⋮\vdots⋮

bn=bn−bb−1b_n = b_n - b_{b-1}bn​=bn​−bb−1​

故 a1=b1,a2=b1+b2,…,an=b1+b2+⋯+bna_1 = b_1,a_2 = b_1 + b_2,\dots,a_n = b_1 + b_2 + \dots + b_na1​=b1​,a2​=b1​+b2​,…,an​=b1​+b2​+⋯+bn​;

即 ai=b1+b2+⋯+bia_i = b_1 + b_2 + \dots + b_iai​=b1​+b2​+⋯+bi​。

其中称 bbb 为 aaa 的差分，aaa 为 bbb 的前缀和，两者是一组逆运算。

# 用途

当要区间更新数列 aaa 的值时（ 序列中 [l,r][l, r][l,r] 之间的每个数加上 ccc ），最简单的做法是遍历数组，在指定区间更新，时间复杂度 O(n)O(n)O(n);

而差分数组可以实现 O(1)O(1)O(1) 的时间复杂度更，公式为 bl+=c,br+1−=cb_l += c, \ b_{r + 1} -= cbl​+=c,br+1​−=c ;

此时 数列 bbb 为：

b1=a1{b_1 = a_1}b1​=a1​

b2=a2−a1b_2 = a_2 - a_1b2​=a2​−a1​

⋮\vdots⋮

bl=bl−bl−1+cb_l = b_l - b_{l-1} + cbl​=bl​−bl−1​+c

⋮\vdots⋮

br+1=br+1−br−cb_{r+1} = b_{r+1} - b_r - cbr+1​=br+1​−br​−c

⋮\vdots⋮

bn=bn−bb−1b_n = b_n - b_{b-1}bn​=bn​−bb−1​

则，数列 aaa 如下：

a1=b1a_1 = b_1a1​=b1​

a2=b1+b2a_2 = b_1 + b_2a2​=b1​+b2​

# 构造

那如何构造数列 aaa, bbb 呢？

将数列 aaa 看作原数列，则 bi=ai−ai−1,a0=0,i≥1b_i = a_i - a_{i-1}, a_0 = 0, i ≥ 1bi​=ai​−ai−1​,a0​=0,i≥1。

int n, m;
int a[N], b[N];

// 区间更新
void insert(int l, int r, int c) {
    b[l] += c;
    b[r + 1] -= c;
}

// 读入数列 a
for (int i = 1; i <= n; i++) scanf("%d", &a[i]);

// 构造数列 b
for (int i = 1; i <= n; i++) b[i] = a[i] - a[i - 1];

// 区间更新：[l...r]元素加 c , 1 ≤ l ≤ r ≤ n
insert(l, r, c);

// 还原数列 a，求前缀和
for (int i = 1; i <= n; i++) a[i] = a[i - 1] + b[i]; // 或 b[i] += b[i - 1]

# 差分矩阵

和差分数列相同，适用于二维数组的区间更新。更新后的原数组通过二维前缀和求解

#include <iostream>

const int N = 1e3 + 7;

int n, m, q, a[N][N], b[N][N];
// 区间更新
void insert(int x1, int y1, int x2, int y2, int c) {
    b[x1][y1] += c;
    b[x2 + 1][y1] -= c;
    b[x1][y2 + 1] -= c;
    b[x2 + 1][y2 + 1] += c;
}
int main() {
    scanf("%d%d%d", &n, &m, &q);
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++) {
            scanf("%d", &a[i][j]); 
            insert(i, j, i, j, a[i][j]); // 构造 b 矩阵
        }

    while (q--) {
        int x1, x2, y1, y2, c;
        scanf("%d%d%d%d%d", &x1, &y1, &x2, &y2, &c);
        insert(x1, y1, x2, y2, c); // 区间更新
    }

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            b[i][j] += b[i - 1][j] + b[i][j - 1] - b[i - 1][j - 1]; //前缀和求更新后矩阵
            printf("%d ", b[i][j]);
        }
        printf("\n");
    }

    return 0;
}
