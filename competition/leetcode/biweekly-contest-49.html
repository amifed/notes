# 2.句子相似性 III (opens new window)

双指针

/**
 * @param {string} sentence1
 * @param {string} sentence2
 * @return {boolean}
 */
var areSentencesSimilar = function(s1, s2) {
    if (s1 == s2) return true
    s1 = s1.split(' ')
    s2 = s2.split(' ')
    const m = s1.length, n = s2.length
    let l1 = 0, l2 = 0
    while (l1 < m && l2 < n && s1[l1] == s2[l2]) ++l1, ++l2
    let r1 = m - 1, r2 = n - 1
    while (r1 >= 0 && r2 >= 0 && s1[r1] == s2[r2]) --r1, --r2
    return l1 > r1 || l2 > r2
};

# 3.统计一个数组中好对子的数目 (opens new window)

nums[i]+rev(nums[j])==nums[j]+rev(nums[i])nums[i]+rev(nums[j])==nums[j]+rev(nums[i])nums[i]+rev(nums[j])==nums[j]+rev(nums[i])

等价转化为

nums[i]−rev(nums[i])==nums[j]−rev(nums[j])nums[i]-rev(nums[i])==nums[j]-rev(nums[j])nums[i]−rev(nums[i])==nums[j]−rev(nums[j])

统计 nums[i]−rev(nums[i])nums[i]-rev(nums[i])nums[i]−rev(nums[i]) 的个数，求两两组合的组合数 =t(t−1)2=\displaystyle{\frac{t(t-1)}{2}}=2t(t−1)​

using ll = long long;
const int mod = 1e9 + 7;
class Solution {
public:
    int rev(int x) {
        int ans = 0;
        while (x) {
            ans = ans * 10 + x % 10;
            x /= 10;
        }
        return ans;
    }
    int countNicePairs(vector<int>& nums) {
        unordered_map<int, int> mp;
        for (int x : nums) ++mp[x - rev(x)];
        ll ans = 0;
        for (auto &p : mp) {
            ll t = p.second;
            ans = (ans + t * (t - 1) / 2) % mod;
        }
        return ans % mod;
    }
};
