# 1.交替合并字符串 (opens new window)

双指针

class Solution {
public:
    string mergeAlternately(string s, string t) {
        string ans;
        int i = 0, j = 0;
        while (i < s.size() || j < t.size()) {
            ans += i < s.size() ? string(1, s[i]) : "";
            ans += j < t.size() ? string(1, t[j]) : "";
            ++i, ++j;
        }
        return ans;
    }
};

# 2.移动所有球到每个盒子所需的最小操作数 (opens new window)

# 解法一 暴力

中心扩展，统计 '1' 的数目和

# 写法一

class Solution {
public:
    vector<int> minOperations(string b) {
        int n = b.size();
        vector<int> ans;
        for (int i = 0; i < n; ++i) {
            int l = 0, r = 0;
            for (int j = i - 1; j >= 0; --j) {
                if (b[j] == '1') l += i - j;
            }
            for (int j = i + 1; j < n; ++j) {
                if (b[j] == '1') r += j - i;
            }
            ans.push_back(l + r);
        }
        return ans;
    }
};

# 写法二

class Solution {
public:
    vector<int> minOperations(string b) {
        int n = b.size();
        vector<int> ans;
        for (int i = 0; i < n; ++i) {
            int t = 0;
            for (int j = 0; j < n; ++j) {
                if (b[j] == '1') t += abs(j - i);
            }
            ans.push_back(t);
        }
        return ans;
    }
};

# 解法二 记录

计算所有的 '1' 到第一个位置的距离之和为 sss； 再次遍历时，设第 iii 个位置左边 '1' 的个数为 lll，右边 '1' 的个数为 rrr，每次移动左边所有的 '1' 距离 iii 的总和多 lll，右边所有的 '1' 距离 iii 的总和少 rrr，即 s+l−rs + l - rs+l−r;

class Solution {
public:
    vector<int> minOperations(string b) {
        int n = b.size();
        vector<int> ans;
        int l = 0, r = 0, s = 0;
        for (int i = 0; i < n; ++i) {
            if (b[i] == '1') ++r, s += i;
        }
        s += r;
        for (int i = 0; i < n; ++i) {
            s = s + l - r;
            ans.push_back(s);
            if (b[i] == '1') ++l, --r;
        }
        return ans;
    }
};
