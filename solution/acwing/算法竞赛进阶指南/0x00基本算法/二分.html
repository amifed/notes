# 最佳牛围栏 (opens new window)

题面：给定正整数数列 AAA，求一个平均数最大的、长度不小于 LLL 的（连续的）字段。

二分答案，判定 “是否存在一个长度不小于 LLL 的子段，平均数不小于二分的值”，即

(a1+a2+⋯+al)l≥mid\displaystyle\frac{(a_1+a_2+\cdots+a_l)}{l}≥midl(a1​+a2​+⋯+al​)​≥mid。

转化问题：将数列中每个数都减去二分的值，就转化为判定 “是否存在一个长度不小于 LLL 的子段，字段和非负”。

(a1+a2+⋯+al)l≥mid\displaystyle\frac{(a_1+a_2+\cdots+a_l)}{l}≥midl(a1​+a2​+⋯+al​)​≥mid

(a1+a2+⋯+al)l−mid≥0\displaystyle\frac{(a_1+a_2+\cdots+a_l)}{l}-mid≥0l(a1​+a2​+⋯+al​)​−mid≥0

(a1−mid+a2−mid+⋯+al−mid)l≥0\displaystyle\frac{(a_1-mid+a_2-mid+\cdots+a_l-mid)}{l}≥0l(a1​−mid+a2​−mid+⋯+al​−mid)​≥0

这里是长度大于等于 LLL，就可以使用前缀和和双指针判定答案。

注意小数二分的写法

#include <iostream>
using namespace std;

const int N = 1e5 + 7;
const double EPS = 1e-5;

int n, m;
int a[N];
double s[N];

bool check(double mid) {
    for (int i = 1; i <= n; ++i) s[i] = s[i - 1] + a[i] - mid;
    double mi = 0;
    for (int i = 0, j = m; j <= n; ++i, ++j) {
        mi = min(mi, s[i]);
        if (s[j] - mi >= 0) return 1;
    }
    return 0;
}

int main() {
    cin >> n >> m;
    double l = 0, r = 0;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        r = max(r, (double)a[i]);
    }
    while (r - l > EPS) {
        double mid = (l + r) / 2;
        if (check(mid)) l = mid;
        else r = mid;
    }
    cout << int(r * 1000);
    return 0;
}

# 特殊排序 (opens new window)

类似于折半插入排序，需要调用 compare 得到大小关系；

// Forward declaration of compare API.
// bool compare(int a, int b);
// return bool means whether a is less than b.

class Solution {
public:
    vector<int> specialSort(int n) {
        vector<int> ans = {1};
        for (int i = 2; i <= n; ++i) {
            int l = 0, r = ans.size() - 1;
            while (l < r) {
                int mid = l + r + 1 >> 1;
                if (compare(ans[mid], i)) l = mid;
                else r = mid - 1;
            }
            ans.push_back(i);
            for (int j = ans.size() - 1; j - 1 > r; --j) swap(ans[j - 1], ans[j]);
            if (compare(i, ans[r])) swap(ans[r], ans[r + 1]);
        }
        return ans;
    }
};
