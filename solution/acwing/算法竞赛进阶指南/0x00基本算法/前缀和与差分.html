# 前缀和

  • 一维前缀和

对于一个给定的数列 AAA ，它的前缀和数列 SSS 定义为：

S[i]=∑j=1iA[j]S[i]=\sum_{j=1}^i A[j] S[i]=j=1∑i​A[j]

求数列一部分和，可表示为：

sum(l,r)=∑i=ljA[i]=S[r]−S[l−1]sum(l,r)=\sum_{i=l}^jA[i]=S[r]-S[l-1] sum(l,r)=i=l∑j​A[i]=S[r]−S[l−1]

# 激光炸弹 (opens new window)

#include <iostream>
using namespace std;

const int N = 5e3 + 7;

int n, m;
int s[N][N];

int main() {
    int T, R;
    cin >> T >> R;
    R = min(5001, R);
    n = m = 5001;
    for (int i = 0, x, y, w; i < T; ++i) {
        cin >> x >> y >> w;
        ++x, ++y;
        s[x][y] += w;
    }

    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= m; ++j) {
            s[i][j] += s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1];
        }
    }

    int ans = 0;
    for (int i = R; i <= n; ++i) {
        for (int j = R; j <= m; ++j) {
            ans = max(ans, s[i][j] - s[i - R][j] - s[i][j - R] + s[i - R][j - R]);
        }
    }
    cout << ans;
    return 0;
}

B[1]=A[1],B[i]=A[i]−A[i−1](2≤i≤n)B[1]=A[1],B[i]=A[i]-A[i-1]\;(2≤i≤n) B[1]=A[1],B[i]=A[i]−A[i−1](2≤i≤n)

# IncDec序列 (opens new window)

求出 aaa 的差分序列 bbb，其中 b1=a1，bi=ai−ai−1(2≤i≤n)b_1=a_1，b_i=a_i-a_{i-1}\;(2≤i≤n)b1​=a1​，bi​=ai​−ai−1​(2≤i≤n)。另 bn+1=0b_{n+1}=0bn+1​=0。

题目中对序列 aaa 的操作，相当与每次可以选出 b1,b2,⋯,bn+1b_1,b_2,\cdots,b_n+1b1​,b2​,⋯,bn​+1 中的任意两个数，一个加 111，一个减 111。 目标是把 b2,b3,⋯,bnb_2,b_3,\cdots,b_nb2​,b3​,⋯,bn​ 变为全零。最终得到的数列 aaa 就是有 nnn 个 b1b_1b1​ 构成的。

#include <iostream>
using namespace std;
using LL = long long;

const int N = 1e5 + 7;
int n;
int a[N], b[N];

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) cin >> a[i];
    for (int i = 1; i <= n; ++i) b[i] = a[i] - a[i - 1];
    LL pos = 0, neg = 0;
    for (int i = 2; i <= n; ++i) {
        if (b[i] > 0) pos += b[i];
        else neg -= b[i];
    }
    cout << min(pos, neg) + abs(pos - neg) << endl;
    cout << abs(pos - neg) + 1;
    return 0;
}

# 最高的牛 (opens new window)

M 对关系实际是牛的相对身高；

首先建一个全为 000 的数组 bbb，若一条关系指明 AiA_iAi​ 和 BiB_iBi​ 可以相互看见（不是一般性，设 Ai<BiA_i<B_iAi​<Bi​），意味着 Ai∼BiA_i\sim B_iAi​∼Bi​ 之间的牛的身高都比二者至少小 111，可以把数组 bbb 中 Ai∼BiA_i \sim B_iAi​∼Bi​ 中的数都减 111。

注意这里操作差分序列不包含两端点，所以是 b[l+1]-=1, b[r]+=1；而不是 b[l]-=1,b[r+1]]+=1

由于第 PPP 头牛是最高的，所以最终 b[P]=0b[P]=0b[P]=0，其他牛与牛 PPP 的身高差就体现在数组 bbb 中，即第 iii 头牛的身高为 H+b[i]H+b[i]H+b[i]。

使用差分的解法可以降低时间复杂度。

#include <iostream>
using namespace std;

const int N = 1e4 + 7;
int n, p, h, m;
bool vt[N][N];
int b[N];

int main() {
    cin >> n >> p >> h >> m;
    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        if (x > y) swap(x, y);
        if (vt[x][y]) continue;
        vt[x][y] = 1;
        b[x + 1] -= 1, b[y] += 1;
    }
    for (int i = 1; i <= n; ++i) {
        b[i] += b[i - 1];
        cout << h + b[i] << endl;
    }
    return 0;
}

# 复杂度分析

  • 时间复杂度 O(N+M)\mathcal{O(N+M)}O(N+M)
    • 差分序列计算 M\mathcal MM 次；
    • 求前缀和计算 N\mathcal NN 次
  • 空间复杂度 O(N2)\mathcal{O(N^2)}O(N2)
    • 存储差分序列长度为 N\mathcal{N}N；
    • 记录重复关系对 N2\mathcal{N^2}N2，如果使用 set 存储则为 O(M)\mathcal{O(M)}O(M)
